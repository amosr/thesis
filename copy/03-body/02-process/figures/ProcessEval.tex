$$
\ruleIN{
  is[c] = @pending@~v
}{
  \ProcBlockShake
        {@pull@~c~x~(l,us)~(l',us')}
        {is}
        {\Sigma}
        {\cdot}
        {l}
        {is[c \mapsto @have@]}
        {(us, x = v)}
}{PullPending}
$$

$$
\ruleIN{

        {\Sigma}
        {\cdot}
        {l'}

        {us'}
}{PullClosed}
$$
